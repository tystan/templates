\newcommand{\boxone}[2]{%
\vskip3pt%
\begin{tikzpicture}\node[boxonetemplate](box){%
\begin{minipage}{0.99\textwidth}%
\bfr\large{\textbf{\textsf{\textcsfou{#1}}}}\efr\vspace{-20pt}
\color{\csthr}{\rule{\textwidth}{0.5mm}}
\textcsfou{#2}
\end{minipage}%
};\end{tikzpicture}%
\vskip3pt%
}


\newcommand{\boxtwo}[2]{%
\vskip3pt%
\begin{tikzpicture}%
\node[boxtwotemplate](box){%
\begin{minipage}{0.99\textwidth}%
\bfr\textbf{\textcsfou{#1}}\efr\vspace{-12pt}
\textcsfou{#2}
\end{minipage}%
};\end{tikzpicture}%
\vskip3pt%
}

  double arrow/.style args={#1}{
    -triangle 60,line width=#1,\csone, % first arrow
    postaction={draw,-triangle 60,\cstwo!50,line width=(#1)/4,
                shorten <=(#1)/1.2,shorten >=2*(#1)/1.2}, % second arrow
  }
}
\newcommand{\doarrow}[3]{%
\begin{textblock}{1}(#1,#2)
\begin{center}
\begin{tikzpicture}
\draw[double arrow=5pt] 
		(0,1.5) coordinate -- (0,0) coordinate;
\node[right] at (0.8,0.8) {\textcsthr{#3}};
\end{tikzpicture}
\vspace{-20pt}
\end{center}
\end{textblock}
}
